# 依赖刷新

假如你的接口需要动态监听某些值的变化而发出请求的，你可以这样处理

import { defineComponent, ref, watch, reactive } from 'vue';
import { useRequest } from 'vue-request';

export default defineComponent({
  name: 'App',
  setup() {
    const someRef = ref(0);
    const someReactive = reactive({
      count: 0,
    });
    const { data, refresh } = useRequest('api/users');

    watch([someRef, () => someReactive.count], refresh);

    return () => <div>{data.value}</div>;
  },
});












 




但是我们也提供了 refreshDeps 。事实上，使用 refreshDeps 的效果和你直接写 watch 监听是没有任何区别的，因此可以理解为 refreshDeps 是上面示例的一个语法糖

提示

切换下面的 radio，会触发请求

import { defineComponent, ref, reactive } from 'vue';
import { useRequest } from 'vue-request';
import Mock from 'mockjs';

const cnameData = Mock.mock({ 'array|5': ['@cname'] });
const nameData = Mock.mock({ 'array|5': ['@name'] });

function testService(type: number): Promise<string[]> {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(type ? nameData.array : cnameData.array);
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const typeRef = ref(0);
    const { data, loading } = useRequest(() => testService(typeRef.value), {
      refreshDeps: [typeRef],
    });

    return () => (
      <div>
        <div>
          <input
            type="radio"
            id="contactChoice1"
            checked={typeRef.value === 0}
            onChange={() => (typeRef.value = 0)}
          />
          <label for="contactChoice1">Chinese Names</label>
          &nbsp; &nbsp;
          <input
            type="radio"
            id="contactChoice2"
            checked={typeRef.value === 1}
            onChange={() => (typeRef.value = 1)}
          />
          <label for="contactChoice2">English Names</label>
        </div>
        <ul>{loading.value ? 'loading...' : data.value!.map((item) => <li>{item}</li>)}</ul>
      </div>
    );
  },
});
