# 错误重试

在应用程序中，暂时性错误其实是很常见的，例如：接口服务器与数据库服务器的连接暂时断开，或者用户端网络暂时发生故障。这些故障往往都会在短时间内自行修正，如果能在延迟合适的时间后再尝试，该请求可能会成功。

现在你可以通过简单的配置，让 VueRequest 帮你处理这件事。只需要提供一个 errorRetryCount 来告诉我们重试的次数。正如下面的例子

# 重试次数

import { defineComponent, ref, watchEffect } from 'vue';
import { useRequest } from 'vue-request';
import Mock from 'mockjs';

let error = true;

function testService() {
  return new Promise((resolve, reject) => {
    if (error) {
      setTimeout(() => {
        reject('Request failed');
      }, 2000);
    } else {
      setTimeout(() => {
        resolve(Mock.mock('@name'));
      }, 1000);
    }
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const errorCount = ref(0);
    const isErrorRequest = ref(error);
    const { data, loading, run } = useRequest(testService, {
      errorRetryCount: 5, // it will retry 5 times
      onError: () => errorCount.value++,
      onSuccess: () => (errorCount.value = 0),
    });

    watchEffect(() => {
      error = isErrorRequest.value;
    });

    return () => (
      <div>
        <button onClick={() => (isErrorRequest.value = !isErrorRequest.value)}>
          switch to <b>{isErrorRequest.value ? 'success' : 'error'}</b> request
        </button>
        &nbsp;&nbsp;&nbsp;
        <button
          onClick={() => {
            errorCount.value = 0;
            run();
          }}
        >
          run
        </button>
        <br />
        <br />
        <div style="height: 32px">{errorCount.value > 0 && `error times: ${errorCount.value}`}</div>
        <br />
        <div style="height: 32px">
          {loading.value ? (errorCount.value ? 'retry loading...' : 'loading...') : data.value}
        </div>
      </div>
    );
  },
});


























 

































# 重试间隔时间

提示

默认情况下，我们将会使用 二进制指数退避算法在新窗口打开 来帮你计算出合适的间隔时间

当然，你也可以提供 errorRetryInterval 来设定重试的间隔时间

import { defineComponent } from 'vue';
import { useRequest } from 'vue-request';

export default defineComponent({
  name: 'App',
  setup() {
    const { data } = useRequest('api/users', {
      errorRetryCount: 5, // it will retry 5 times
      errorRetryInterval: 3000, // The retry interval is 5 seconds
    });
    return () => <div>{data.value}</div>;
  },
});
