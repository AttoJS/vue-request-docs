# 屏幕聚焦重新请求

有些时候，你要确保多个浏览器窗口之间数据的一致性；又或者是当用户电脑在休眠状态重新激活后，页面的数据需要同步到最新状态时。refreshOnWindowFocus 可能会为你节省很多逻辑代码，如下所示，你只需要启用 refreshOnWindowFocus 即可

提示

你可以尝试切换窗口或者隐藏窗口然后再切换回来，以观察下面的例子（默认时间间隔为 5 秒）

import { defineComponent } from 'vue';
import { useRequest } from 'vue-request';
import Mock from 'mockjs';

function testService() {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(Mock.mock('@name'));
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { data, loading } = useRequest(testService, {
      refreshOnWindowFocus: true,
    });
    return () => <div>random name: {loading.value ? 'loading...' : data.value}</div>;
  },
});
















 




# 重新聚焦间隔时间

如果你不希望用户在短时间内频繁触发 refreshOnWindowFocus 导致频繁请求的话，你可以修改 refocusTimespan 的值，从而延长触发的间隔，如下所示

import { defineComponent } from 'vue';
import { useRequest } from 'vue-request';
import Mock from 'mockjs';

function testService() {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(Mock.mock('@name'));
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { data } = useRequest(testService, {
      refreshOnWindowFocus: true,
      refocusTimespan: 2000, // 2000ms
    });
    return () => <div>random name: {loading.value ? 'loading...' : data.value}</div>;
  },
});
